create table user values(username varchar(20) primary key not null,password varchar(255), name varchar(30) default null,bio varchar(255),dp varchar(255),private boolean);

create table notifications(username varchar(20) references user(username) on delete cascade,type integer(1),receiver varchar(20) references user(username) on delete cascade, primary key(receiver),created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);

create table post(username varchar(20) references user(username) on delete cascade, postid integer(10) primary key auto_increment, img varchar(255),caption varchar(255),private boolean,created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);

create table comments(postid integer(10) references post(postid) on delete cascade,username varchar(20) references user(username) on delete cascade,comment varchar(255),created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);

create table likes(postid integer(10) references post(postid) on delete cascade,username varchar(20) references user(username) on delete cascade);

create table followers(username varchar(20) references user(username) on delete cascade,follower varchar(20) references user(username) on delete cascade);

create table requests(to varchar(20) references user(username) on delete cascade,from varchar(20) references user(username) on delete cascade,created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
